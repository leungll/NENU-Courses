f=imread('pout.tif');
f1=fspecial('laplacian',0);
f2=im2double(f);
f3=imfilter(f2,f1);
ff=f2-f3;
f11=fspecial('laplacian',1);
f33=imfilter(f2,f11);
ff1=f2-f33;
subplot(1,4,1),imshow(ff),title('laplacian,0');
subplot(1,4,2),imshow(ff1),title('laplacian,1');
g=imread('pout.tif');
g1=fspecial('sobel');
g2=imfilter(g,g1);
g3=g-g2;
subplot(1,4,3),imshow(g3),title('sobel');
